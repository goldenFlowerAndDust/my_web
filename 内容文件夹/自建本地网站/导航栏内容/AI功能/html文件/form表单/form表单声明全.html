form 全部声明速查表（中文含义 + 作用 + 演示）

声明中文含义作用简述一行演示
action提交地址数据要发送到哪个 URL<form action="/search">
method提交方法get（查询）或 post（写入）<form method="post">
enctype编码类型文件上传时必须 multipart/form-data<form enctype="multipart/form-data">
target目标窗口_blank 新标签打开；_self 默认<form target="_blank">
autocomplete自动完成on 启用（默认）或 off 关闭<form autocomplete="off">
novalidate不验证禁用浏览器原生校验（纯前端）<form novalidate>
type="text"单行文本普通字符串输入<input type="text" name="user">
type="password"密码掩码输入<input type="password" name="pwd">
type="email"邮箱自带 @ 格式校验<input type="email" name="mail">
type="tel"电话唤起数字键盘（移动端）<input type="tel" name="phone">
type="url"网址自带 URL 格式校验<input type="url" name="site">
type="number"数字仅允许数值，可设 min/max/step<input type="number" min="0" max="100" step="5">
type="range"滑块可视化数值选择<input type="range" min="0" max="100" value="50">
type="date"日期自带日期选择器<input type="date" name="birth">
type="time"时间HH:mm 选择器<input type="time" name="alarm">
type="datetime-local"本地日期时间日期+时间（无时区）<input type="datetime-local" name="meet">
type="month"月份YYYY-MM 选择器<input type="month" name="exp">
type="week"周YYYY-W## 选择器<input type="week" name="week">
type="color"颜色原生取色器<input type="color" name="theme">
type="checkbox"复选框多选 on/off<input type="checkbox" name="hobby" value="music">
type="radio"单选框name 相同组内互斥<input type="radio" name="gender" value="f">
type="file"文件上传选择；需 enctype 支持<input type="file" name="avatar" accept="image/*">
type="hidden"隐藏域不显示，只传数据<input type="hidden" name="csrf" value="abc123">
type="submit"提交按钮触发表单提交<input type="submit" value="保存">
type="reset"重置按钮清空表单内容<input type="reset" value="清空">
type="button"普通按钮无默认行为，纯 JS 用<input type="button" value="更多" onclick="add()">
name字段名键名，后端接收用<input name="username">
value初始值默认文本或选项值<input value="游客">
placeholder占位提示灰字提示，不提交<input placeholder="请输入昵称">
required必填阻止空值提交（原生校验）<input required>
disabled禁用灰掉，不提交<input disabled>
readonly只读可查看，不可改，会提交<input readonly>
maxlength最大长度字符上限<input maxlength="20">
min / max最小/最大值数字、日期等范围<input type="number" min="0" max="100">
step步长数字/滑块每次跳跃量<input type="number" step="0.1">
pattern正则校验原生格式检查<input pattern="\d{4}">（4 位数字）
autofocus自动聚焦进入页面即聚焦该字段<input autofocus>
autocomplete自动完成on / off 浏览器记忆<input autocomplete="off">
accept接受类型文件上传时限定 MIME<input type="file" accept="image/*">
multiple多选/多文件文件或 email 可多值<input type="file" multiple>
<textarea>多行文本可换行的大文本框<textarea rows="4" cols="40"></textarea>
<select>下拉选择单选或多选（multiple）<select><option value="a">A</option></select>
<label>标签点击文字=点击控件；for 关联 ID<label for="u">用户名</label><input id="u">
<fieldset>字段分组 visually 分组 + legend 标题<fieldset><legend>登录</legend> … </fieldset>
<datalist>数据列表input 联想下拉（兼容）<input list="city"><datalist id="city"><option value="Beijing"></datalist>
<output>输出结果纯展示计算结果（JS 写 value）<output name="res">0</output>
